
```powershell
tasklist /svc # Enumerate all services

Get-Process | Where-Object { $_.Name -notin @('smss', 'csrss', 'winlogon', 'lsass', 'svchost', 'services', 'wininit', 'System', 'Idle','RuntimeBroker','RuntimeBroker','WmiPrvSE','dllhost','backgroundTaskHost','InstallAgent','taskhostw','vmtoolsd','explorer') } | Select-Object Name, Id

Get-WmiObject Win32_Service | Where-Object { $_.Name -notin @('EventLog', 'Winmgmt', 'wuauserv', 'Dhcp', 'Dnscache', 'LanmanServer', 'LanmanWorkstation', 'PlugPlay', 'Power') } | Select-Object Name, DisplayName, ProcessId, State, StartMode

# cmd
tasklist /svc | findstr /v /i "smss.exe csrss.exe winlogon.exe lsass.exe svchost.exe services.exe wininit.exe"
```
